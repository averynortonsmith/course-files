  \item 10N.1.sl.TZ0.2\\
  Let $g(x)=2x \sin x$.

  \item 12M.1.sl.TZ1.3\\
  Let $f(x)=e^{6x}$.
  \begin{enumerate}
    \item Write down $f'(x)$ [1 mark]
    \item The tangent to the graph of $f$ at the point $P(0,b)$ has gradient $m$.  [4 marks]
    \begin{enumerate}
      \item Show that $m=6$.
      \item Find $b$.
    \end{enumerate}
    \item Hence, write down the equation of this tangent. [1 mark]
  \end{enumerate}
